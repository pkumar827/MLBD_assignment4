import re
import math
from collections import defaultdict

STOP_WORDS = {"a","an","the","they","these","this","for","is","are","was","of","or","and","does","will","whose"}


class InvertedIndex:
    def __init__(self):
        self.index = defaultdict(list)
        self.documents = {}

    def preprocess(self, text):
        text = text.lower()
        text = re.sub(r'[^\w\s]', ' ', text)
        words = text.split()
        return [w for w in words if w not in STOP_WORDS]

    def add_document(self, doc_id, text):
        words = self.preprocess(text)
        self.documents[doc_id] = words
        
        for pos, word in enumerate(words):
            self.index[word].append((doc_id, pos))

    def tf(self, word, doc_id):
        words = self.documents[doc_id]
        return words.count(word) / len(words)

    def idf(self, word):
        N = len(self.documents)
        count = len(set(doc for doc, _ in self.index[word]))
        return math.log(N / (count + 1))

    def search(self, query):
        query_words = self.preprocess(query)
        scores = defaultdict(float)

        for word in query_words:
            if word in self.index:
                for doc, _ in self.index[word]:
                    scores[doc] += self.tf(word, doc) * self.idf(word)

        return sorted(scores.items(), key=lambda x: x[1], reverse=True)
